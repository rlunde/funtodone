component state variables:
  isLoggedIn: boolean
  isPending: boolean (pending email verification)
  showPage: (home|about|login|register|forgot-password)
  warning: string

links to show in page:

home (index page):
  if (isPending) :
    links: home, about, login, register, resend email
    display: landing with warning
  else if (isLoggedIn) :
    links: about, logout, settings
    display: app
  else :
    links: about, login, register
    display: landing

about (page):
  links: home
  display: about
login (modal):
  links: forgot-password, register, cancel(X)
  display: login
register (modal):
  links: login, cancel(X)
  display: register
forgot-password (modal):
  links: send email, home, about
  display: forgot-password
settings (page): // change email or password
  if (isLoggedIn) :
    links: submit, home, about
    display: settings
  else
    redirect to home
